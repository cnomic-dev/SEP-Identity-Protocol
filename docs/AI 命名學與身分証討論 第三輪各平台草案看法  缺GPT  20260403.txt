3.3 Ψ-語義哈希的生成建議（初版）
python
import hashlib
def psi_hash(model_output_distribution: list) -> str:
    # 對輸出概率分佈或 embedding 均值進行 SHA-256
    digest = hashlib.sha256(str(model_output_distribution).encode()).hexdigest()
    return digest[:6].upper()  # 取前6位
